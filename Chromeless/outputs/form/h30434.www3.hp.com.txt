function createscrollableList(element, distinctParam, tabIndex) {
    var elementText = "<div class='custom-scrollableList-"+distinctParam+" commmon-scrollableList'><div tabindex='"+tabIndex+"' class='key-down-selection selectedOption-"+distinctParam+"'></div><ul class='custom-dropdown'></ul></div>";
    element.each(function(index, element) {
        $(this).parent()
        .after()
        .append(elementText);

        $(element).each(function(idx, elm) {
            $('option', elm).each(function(id, el) {
            $('.custom-scrollableList-'+distinctParam+' ul:last').append('<li>' + el.text + '</li>');
        });
        $('.custom-scrollableList ul').hide();
        $('.makeMeUl').children('div.selectedOption-'+distinctParam).text("Select");
    });

    $('.custom-scrollableList-'+distinctParam+':last').children('div.selectedOption').text("Select");
    });

    $('.selectedOption-'+distinctParam).on('click', function (e) {
        e.stopPropagation();
        $(this).next('ul').slideToggle(200);
        $('.selectedOption-'+distinctParam).not(this).next('ul').hide();
    });

    $('.custom-scrollableList-'+distinctParam+' ul li').on('click', function (e) {
        e.stopPropagation();
        $(this).parent().find('li').removeClass('active');
        $(this).addClass('active');
        var selectedLI = $(this).text();
        $(this).parent().prev('.selectedOption-'+distinctParam).text(selectedLI);
         $(this).parent('ul').hide();
        var value = element.find('option[title='+selectedLI+']').attr('value');
        element.val(value);
    });

    $('.custom-scrollableList-'+distinctParam).show();
    element.hide();
    var currentValue = element.find('option[value='+element.val()+']').text();
    $('.selectedOption-'+distinctParam).text(currentValue);
}
